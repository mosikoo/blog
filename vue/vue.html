5.9 Vue

lifecycle

computed vs methods

var a = new Vue({
  data: {
    message: 'test'
  },
  computed: {
    computeMessage: function() {
      return this.message + 'aaa'; // 有缓存
    }
  },
  methods: {
    computeMessage: function() {
      return this.message + 'aaa'; // render 触发
    }
  }
})

watche

var vm = new Vue({
  el: '#demo',
  data: {
    firstName: 'Foo',
    lastName: 'Bar',
    fullName: 'Foo Bar'
  },
  watch: {
    firstName: function (val) {
      this.fullName = val + ' ' + this.lastName
    },
    lastName: function (val) {
      this.fullName = this.firstName + ' ' + val
    }
  }
})

Vuex

const store = new Vuex.Store({
  state: {
    count: 0
  },
  // 监听 mutation 必须是同步函数
  mutations: {
    increment (state, payload) {
      state.count++
    }
  },
  // 用于快速获取处理后的变量
  getters: {
    countGetter: state => {
      return state.count++
    }
  },
  // 异步处理
  actions: {
    increment (context, payload) { //  context类似与store
      context.commit('increment')
    }
  }
})
// 触发mutations
store.commit('increment')
// 触发actions
store.dispatch('increment')
